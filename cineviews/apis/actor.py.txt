import traceback
from flask import Blueprint, jsonify
from main.database import Session
from cine.models import Actor

api = Blueprint('cine_apis_actors', __name__)


@api.route('/apis/v1/actors', methods=['GET'])
def fetch_all_actors():
    response = None
    status = 200

    session = Session()

    try:
        actors = session.query(Actor).all()

        response = jsonify({
            "message": "lista de actores",
            "data": [actor.to_dict() for actor in actors],
            "success": True,
            "error": None
        })

    except Exception as e:
        traceback.print_exc()

        response = jsonify({
            "message": "Ocurrió un error al listar actores",
            "data": None,
            "success": False,
            "error": str(e)
        })
        status = 500

    finally:
        session.close()

    return response, status
